pnat.xgcd_type :
Type

A term of xgcd_type is a system of six naturals. They should be thought of as representing the matrix [[w, x], [y, z]] = [[wp + 1, x], [y, zp + 1]] together with the vector [a, b] = [ap + 1, bp + 1].

pnat.gcd_a'_coe (a b : ℕ+) :
↑(pnat.gcd_a' a b) = ↑(pnat.gcd_w a b) + pnat.gcd_x a b

pnat.gcd_b'_coe (a b : ℕ+) :
↑(pnat.gcd_b' a b) = pnat.gcd_y a b + ↑(pnat.gcd_z a b)

pnat.gcd_props (a b d w : ℕ+) (x y : ℕ) (z a' b' : ℕ+) :
w * z = nat.succ_pnat (x * y) ∧ a = a' * d ∧ b = b' * d ∧ z * a' = nat.succ_pnat (x * ↑b') ∧ w * b' = nat.succ_pnat (y * ↑a') ∧ ↑z * ↑a = x * ↑b + ↑d ∧ ↑w * ↑b = y * ↑a + ↑d

pnat.gcd_eq (a b : ℕ+) :
pnat.gcd_d a b = pnat.gcd a b

pnat.gcd_det_eq (a b : ℕ+) :
pnat.gcd_w a b * pnat.gcd_z a b = nat.succ_pnat (pnat.gcd_x a b * pnat.gcd_y a b)

pnat.gcd_a_eq (a b : ℕ+) :
a = pnat.gcd_a' a b * pnat.gcd a b

pnat.gcd_b_eq (a b : ℕ+) :
b = pnat.gcd_b' a b * pnat.gcd a b

pnat.gcd_rel_left' (a b : ℕ+) :
pnat.gcd_z a b * pnat.gcd_a' a b = nat.succ_pnat (pnat.gcd_x a b * ↑(pnat.gcd_b' a b))

pnat.gcd_rel_right' (a b : ℕ+) :
pnat.gcd_w a b * pnat.gcd_b' a b = nat.succ_pnat (pnat.gcd_y a b * ↑(pnat.gcd_a' a b))

pnat.gcd_rel_left (a b : ℕ+) :
↑(pnat.gcd_z a b) * ↑a = pnat.gcd_x a b * ↑b + ↑(pnat.gcd a b)

pnat.gcd_rel_right (a b : ℕ+) :
↑(pnat.gcd_w a b) * ↑b = pnat.gcd_y a b * ↑a + ↑(pnat.gcd a b)
