pairwise {α : Type u_1} (p : α → α → Prop) :
Prop

A relation p holds pairwise if p i j for all i ≠ j.

pairwise_on_bool {α : Type u} {r : α → α → Prop} (hr : symmetric r) {a b : α} :
pairwise (r on λ (c : bool), cond c a b) ↔ r a b

pairwise_disjoint_on_bool {α : Type u} [lattice.semilattice_inf_bot α] {a b : α} :
pairwise (disjoint on λ (c : bool), cond c a b) ↔ disjoint a b

set.disjointed {α : Type u} (f : ℕ → set α) (n : ℕ) :
set α

If f : ℕ → set α is a sequence of sets, then disjointed f is the sequence formed with each set subtracted from the later ones in the sequence, to form a disjoint sequence.

set.disjoint_disjointed {α : Type u} {f : ℕ → set α} :
pairwise (disjoint on set.disjointed f)

set.disjoint_disjointed' {α : Type u} {f : ℕ → set α} (i j : ℕ) (a : i ≠ j) :
set.disjointed f i ∩ set.disjointed f j = ∅

set.disjointed_subset {α : Type u} {f : ℕ → set α} {n : ℕ} :
set.disjointed f n ⊆ f n

set.Union_lt_succ {α : Type u} {f : ℕ → set α} {n : ℕ} :
(⋃ (i : ℕ) (H : i < nat.succ n), f i) = f n ∪ ⋃ (i : ℕ) (H : i < n), f i

set.Inter_lt_succ {α : Type u} {f : ℕ → set α} {n : ℕ} :
(⋂ (i : ℕ) (H : i < nat.succ n), f i) = f n ∩ ⋂ (i : ℕ) (H : i < n), f i

set.Union_disjointed {α : Type u} {f : ℕ → set α} :
(⋃ (n : ℕ), set.disjointed f n) = ⋃ (n : ℕ), f n

set.disjointed_induct {α : Type u} {f : ℕ → set α} {n : ℕ} {p : set α → Prop} (h₁ : p (f n)) (h₂ : ∀ (t : set α) (i : ℕ), p t → p (t \ f i)) :
p (set.disjointed f n)

set.disjointed_of_mono {α : Type u} {f : ℕ → set α} {n : ℕ} (hf : monotone f) :
set.disjointed f (n + 1) = f (n + 1) \ f n

set.Union_disjointed_of_mono {α : Type u} {f : ℕ → set α} (hf : monotone f) (n : ℕ) :
(⋃ (i : ℕ) (H : i < nat.succ n), set.disjointed f i) = f n
