computation.map_parallel {α : Type u} {β : Type v} (f : α → β) (S : wseq (computation α)) :
computation.map f (computation.parallel S) = computation.parallel (wseq.map (computation.map f) S)

computation.parallel_empty {α : Type u} (S : wseq (computation α)) (h : wseq.head S ~> option.none) :
