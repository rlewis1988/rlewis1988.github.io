irrational

ℝ → Prop

irr_nrt_of_notint_nrt

∀ {x : ℝ} (n : ℕ) (m : ℤ), x ^ n = ↑m → (¬∃ (y : ℤ), x = ↑y) → 0 < n → irrational x

irr_nrt_of_n_not_dvd_multiplicity

∀ {x : ℝ} (n : ℕ) {m : ℤ} (hm : m ≠ 0) (p : ℕ) [hp : nat.prime p], x ^ n = ↑m → (multiplicity ↑p m).get _ % n ≠ 0 → irrational x

irr_sqrt_of_multiplicity_odd

∀ (m : ℤ) (hm : 0 < m) (p : ℕ) [hp : nat.prime p], (multiplicity ↑p m).get _ % 2 = 1 → irrational (real.sqrt ↑m)

irr_sqrt_of_prime

∀ (p : ℕ), nat.prime p → irrational (real.sqrt ↑p)

irr_sqrt_two

irrational (real.sqrt 2)

irr_sqrt_rat_iff

∀ (q : ℚ), irrational (real.sqrt ↑q) ↔ rat.sqrt q * rat.sqrt q ≠ q ∧ 0 ≤ q

irrational.decidable

Π (q : ℚ), decidable (irrational (real.sqrt ↑q))

Attributes: instance

irr_rat_add_of_irr

∀ {q : ℚ} {x : ℝ}, irrational x → irrational (↑q + x)

irr_rat_add_iff_irr

∀ {q : ℚ} {x : ℝ}, irrational (↑q + x) ↔ irrational x

Attributes: simp

irr_add_rat_iff_irr

∀ {q : ℚ} {x : ℝ}, irrational (x + ↑q) ↔ irrational x

Attributes: simp

irr_mul_rat_iff_irr

∀ {q : ℚ} {x : ℝ}, q ≠ 0 → (irrational (x * ↑q) ↔ irrational x)

irr_of_irr_mul_self

∀ {x : ℝ}, irrational (x * x) → irrational x

irr_neg

∀ {x : ℝ}, irrational (-x) ↔ irrational x
