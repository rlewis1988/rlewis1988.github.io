cardinal.cantor_function_aux (c : ℝ) (f : ℕ → bool) (n : ℕ) :
ℝ

cardinal.cantor_function_aux_tt {c : ℝ} {f : ℕ → bool} {n : ℕ} (h : f n = bool.tt) :
cardinal.cantor_function_aux c f n = c ^ n

Attributes: simp
cardinal.cantor_function_aux_ff {c : ℝ} {f : ℕ → bool} {n : ℕ} (h : f n = bool.ff) :
cardinal.cantor_function_aux c f n = 0

Attributes: simp
cardinal.cantor_function_aux_nonneg {c : ℝ} {f : ℕ → bool} {n : ℕ} (h : 0 ≤ c) :
0 ≤ cardinal.cantor_function_aux c f n

cardinal.cantor_function_aux_eq {c : ℝ} {f g : ℕ → bool} {n : ℕ} (h : f n = g n) :
cardinal.cantor_function_aux c f n = cardinal.cantor_function_aux c g n

cardinal.cantor_function_aux_succ {c : ℝ} (f : ℕ → bool) :
(λ (n : ℕ), cardinal.cantor_function_aux c f (n + 1)) = λ (n : ℕ), c * cardinal.cantor_function_aux c (λ (n : ℕ), f (n + 1)) n

cardinal.summable_cantor_function {c : ℝ} (f : ℕ → bool) (h1 : 0 ≤ c) (h2 : c < 1) :
summable (cardinal.cantor_function_aux c f)

cardinal.cantor_function (c : ℝ) (f : ℕ → bool) :
ℝ

cardinal.cantor_function_le {c : ℝ} {f g : ℕ → bool} (h1 : 0 ≤ c) (h2 : c < 1) (h3 : ∀ (n : ℕ), ↥(f n) → ↥(g n)) :
cardinal.cantor_function c f ≤ cardinal.cantor_function c g

cardinal.cantor_function_succ {c : ℝ} (f : ℕ → bool) (h1 : 0 ≤ c) (h2 : c < 1) :
cardinal.cantor_function c f = cond (f 0) 1 0 + c * cardinal.cantor_function c (λ (n : ℕ), f (n + 1))

cardinal.increasing_cantor_function {c : ℝ} (h1 : 0 < c) (h2 : c < 1 / 2) {n : ℕ} {f g : ℕ → bool} (hn : ∀ (k : ℕ), k < n → f k = g k) (fn : f n = bool.ff) (gn : g n = bool.tt) :
cardinal.cantor_function c f < cardinal.cantor_function c g

cardinal.injective_cantor_function {c : ℝ} (h1 : 0 < c) (h2 : c < 1 / 2) :
function.injective (cardinal.cantor_function c)
