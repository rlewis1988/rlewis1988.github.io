cardinal.cantor_function_aux_tt

∀ {c : ℝ} {f : ℕ → bool} {n : ℕ}, f n = bool.tt → cardinal.cantor_function_aux c f n = c ^ n

cardinal.cantor_function_aux_ff

cardinal.cantor_function_aux_nonneg

cardinal.cantor_function_aux_eq

∀ {c : ℝ} (f : ℕ → bool), (λ (n : ℕ), cardinal.cantor_function_aux c f (n + 1)) = λ (n : ℕ), c * cardinal.cantor_function_aux c (λ (n : ℕ), f (n + 1)) n

cardinal.cantor_function

ℝ → (ℕ → bool) → ℝ

cardinal.cantor_function_le

∀ {c : ℝ} {f g : ℕ → bool}, 0 ≤ c → c < 1 → (∀ (n : ℕ), ↥(f n) → ↥(g n)) → cardinal.cantor_function c f ≤ cardinal.cantor_function c g

cardinal.cantor_function_succ

∀ {c : ℝ} (f : ℕ → bool), 0 ≤ c → c < 1 → cardinal.cantor_function c f = cond (f 0) 1 0 + c * cardinal.cantor_function c (λ (n : ℕ), f (n + 1))

cardinal.increasing_cantor_function

∀ {c : ℝ}, 0 < c → c < 1 / 2 → ∀ {n : ℕ} {f g : ℕ → bool}, (∀ (k : ℕ), k < n → f k = g k) → f n = bool.ff → g n = bool.tt → cardinal.cantor_function c f < cardinal.cantor_function c g

cardinal.injective_cantor_function

∀ {c : ℝ}, 0 < c → c < 1 / 2 → function.injective (cardinal.cantor_function c)
