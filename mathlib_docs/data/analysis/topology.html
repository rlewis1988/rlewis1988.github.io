A ctop α σ is a realization of a topology (basis) on α, represented by a type σ together with operations for the top element and the intersection operation.

⇑{f := f, top := T, top_mem := h₁, inter := I, inter_mem := h₂, inter_sub := h₃} a = f a

A ctop realizer for the topological space T is a ctop which generates T.

ctop.to_realizer {α : Type u_1} {σ : Type u_3} (F : ctop α σ) :
ctop.realizer α

ctop.realizer.is_basis {α : Type u_1} [T : topological_space α] (F : ctop.realizer α) :
topological_space.is_topological_basis (set.range ((F.F).f))

ctop.realizer.mem_nhds {α : Type u_1} [T : topological_space α] (F : ctop.realizer α) {s : set α} {a : α} :
s ∈ nhds a ↔ ∃ (b : F.σ), a ∈ ⇑(F.F) b ∧ ⇑(F.F) b ⊆ s

ctop.realizer.is_open_iff {α : Type u_1} [topological_space α] (F : ctop.realizer α) {s : set α} :
is_open s ↔ ∀ (a : α), a ∈ s → (∃ (b : F.σ), a ∈ ⇑(F.F) b ∧ ⇑(F.F) b ⊆ s)

ctop.realizer.is_closed_iff {α : Type u_1} [topological_space α] (F : ctop.realizer α) {s : set α} :
is_closed s ↔ ∀ (a : α), (∀ (b : F.σ), a ∈ ⇑(F.F) b → (∃ (z : α), z ∈ ⇑(F.F) b ∩ s)) → a ∈ s

ctop.realizer.mem_interior_iff {α : Type u_1} [topological_space α] (F : ctop.realizer α) {s : set α} {a : α} :
a ∈ interior s ↔ ∃ (b : F.σ), a ∈ ⇑(F.F) b ∧ ⇑(F.F) b ⊆ s

ctop.realizer.is_open {α : Type u_1} [topological_space α] (F : ctop.realizer α) (s : F.σ) :
is_open (⇑(F.F) s)
