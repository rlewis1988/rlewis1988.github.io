nat.sqrt_aux

ℕ → ℕ → ℕ → ℕ

nat.sqrt

ℕ → ℕ

sqrt n is the square root of a natural number n. If n is not a perfect square, it returns the largest k:ℕ such that k*k ≤ n.

nat.sqrt_aux_0

∀ (r n : ℕ), nat.sqrt_aux 0 r n = r

nat.sqrt_aux_1

∀ {r n b : ℕ}, b ≠ 0 → ∀ {n' : ℕ}, r + b + n' = n → nat.sqrt_aux b r n = nat.sqrt_aux (nat.shiftr b 2) (nat.div2 r + b) n'

nat.sqrt_aux_2

∀ {r n b : ℕ}, b ≠ 0 → n < r + b → nat.sqrt_aux b r n = nat.sqrt_aux (nat.shiftr b 2) (nat.div2 r) n

nat.sqrt_le

∀ (n : ℕ), nat.sqrt n * nat.sqrt n ≤ n

nat.lt_succ_sqrt

∀ (n : ℕ), n < nat.succ (nat.sqrt n) * nat.succ (nat.sqrt n)

nat.sqrt_le_add

∀ (n : ℕ), n ≤ nat.sqrt n * nat.sqrt n + nat.sqrt n + nat.sqrt n

nat.le_sqrt

∀ {m n : ℕ}, m ≤ nat.sqrt n ↔ m * m ≤ n

nat.sqrt_lt

∀ {m n : ℕ}, nat.sqrt m < n ↔ m < n * n

nat.sqrt_le_self

∀ (n : ℕ), nat.sqrt n ≤ n

nat.sqrt_le_sqrt

∀ {m n : ℕ}, m ≤ n → nat.sqrt m ≤ nat.sqrt n

nat.sqrt_eq_zero

∀ {n : ℕ}, nat.sqrt n = 0 ↔ n = 0

nat.eq_sqrt

∀ {n q : ℕ}, q = nat.sqrt n ↔ q * q ≤ n ∧ n < (q + 1) * (q + 1)

nat.le_three_of_sqrt_eq_one

∀ {n : ℕ}, nat.sqrt n = 1 → n ≤ 3

nat.sqrt_lt_self

∀ {n : ℕ}, 1 < n → nat.sqrt n < n

nat.sqrt_pos

∀ {n : ℕ}, 0 < nat.sqrt n ↔ 0 < n

nat.sqrt_add_eq

∀ (n : ℕ) {a : ℕ}, a ≤ n + n → nat.sqrt (n * n + a) = n

nat.sqrt_eq

∀ (n : ℕ), nat.sqrt (n * n) = n

nat.sqrt_succ_le_succ_sqrt

∀ (n : ℕ), nat.sqrt (nat.succ n) ≤ nat.succ (nat.sqrt n)

nat.exists_mul_self

∀ (x : ℕ), (∃ (n : ℕ), n * n = x) ↔ nat.sqrt x * nat.sqrt x = x
