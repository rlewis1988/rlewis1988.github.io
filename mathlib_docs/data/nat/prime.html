nat.prime

ℕ → Prop

prime p means that p is a prime number, that is, a natural number at least 2 whose only divisors are p and 1.

nat.prime.two_le

∀ {p : ℕ}, nat.prime p → 2 ≤ p

nat.prime.one_lt

∀ {p : ℕ}, nat.prime p → 1 < p

nat.prime.ne_one

∀ {p : ℕ}, nat.prime p → p ≠ 1

nat.prime_def_lt

∀ {p : ℕ}, nat.prime p ↔ 2 ≤ p ∧ ∀ (m : ℕ), m < p → m ∣ p → m = 1

nat.prime_def_lt'

∀ {p : ℕ}, nat.prime p ↔ 2 ≤ p ∧ ∀ (m : ℕ), 2 ≤ m → m < p → ¬m ∣ p

nat.prime_def_le_sqrt

∀ {p : ℕ}, nat.prime p ↔ 2 ≤ p ∧ ∀ (m : ℕ), 2 ≤ m → m ≤ nat.sqrt p → ¬m ∣ p

nat.factors

ℕ → list ℕ

factors n is the prime factorization of n, listed in increasing order.

nat.mem_factors

∀ {n p : ℕ}, p ∈ nat.factors n → nat.prime p

nat.prod_factors

∀ {n : ℕ}, 0 < n → list.prod (nat.factors n) = n

nat.prime.coprime_iff_not_dvd

∀ {p n : ℕ}, nat.prime p → (nat.coprime p n ↔ ¬p ∣ n)

nat.prime.dvd_iff_not_coprime

∀ {p n : ℕ}, nat.prime p → (p ∣ n ↔ ¬nat.coprime p n)

nat.prime.dvd_mul

∀ {p m n : ℕ}, nat.prime p → (p ∣ m * n ↔ p ∣ m ∨ p ∣ n)

nat.prime.not_dvd_mul

∀ {p m n : ℕ}, nat.prime p → ¬p ∣ m → ¬p ∣ n → ¬p ∣ m * n

nat.prime.dvd_of_dvd_pow

∀ {p m n : ℕ}, nat.prime p → p ∣ m ^ n → p ∣ m

nat.prime.mul_eq_prime_pow_two_iff

∀ {x y p : ℕ}, nat.prime p → x ≠ 1 → y ≠ 1 → (x * y = p ^ 2 ↔ x = p ∧ y = p)

nat.prime.dvd_fact

∀ {n p : ℕ}, nat.prime p → (p ∣ nat.fact n ↔ p ≤ n)

nat.prime.coprime_pow_of_not_dvd

∀ {p m a : ℕ}, nat.prime p → ¬p ∣ a → nat.coprime a (p ^ m)

nat.coprime_primes

∀ {p q : ℕ}, nat.prime p → nat.prime q → (nat.coprime p q ↔ p ≠ q)

nat.coprime_pow_primes

∀ {p q : ℕ} (n m : ℕ), nat.prime p → nat.prime q → p ≠ q → nat.coprime (p ^ n) (q ^ m)

nat.coprime_or_dvd_of_prime

∀ {p : ℕ}, nat.prime p → ∀ (i : ℕ), nat.coprime p i ∨ p ∣ i

nat.dvd_prime_pow

∀ {p : ℕ}, nat.prime p → ∀ {m i : ℕ}, i ∣ p ^ m ↔ ∃ (k : ℕ) (H : k ≤ m), i = p ^ k

nat.mem_list_primes_of_dvd_prod

∀ {p : ℕ}, nat.prime p → ∀ {l : list ℕ}, (∀ (p : ℕ), p ∈ l → nat.prime p) → p ∣ list.prod l → p ∈ l

nat.mem_factors_iff_dvd

∀ {n p : ℕ}, 0 < n → nat.prime p → (p ∈ nat.factors n ↔ p ∣ n)

nat.perm_of_prod_eq_prod

∀ {l₁ l₂ : list ℕ}, list.prod l₁ = list.prod l₂ → (∀ (p : ℕ), p ∈ l₁ → nat.prime p) → (∀ (p : ℕ), p ∈ l₂ → nat.prime p) → l₁ ~ l₂

nat.factors_unique

∀ {n : ℕ} {l : list ℕ}, list.prod l = n → (∀ (p : ℕ), p ∈ l → nat.prime p) → l ~ nat.factors n

nat.succ_dvd_or_succ_dvd_of_succ_sum_dvd_mul

∀ {p : ℕ}, nat.prime p → ∀ {m n k l : ℕ}, p ^ k ∣ m → p ^ l ∣ n → p ^ (k + l + 1) ∣ m * n → p ^ (k + 1) ∣ m ∨ p ^ (l + 1) ∣ n
