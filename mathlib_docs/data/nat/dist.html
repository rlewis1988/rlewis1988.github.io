root/data/nat/dist

nat.dist (n m : ℕ) :
ℕ

Distance (absolute value of difference) between natural numbers.

nat.dist.def (n m : ℕ) :
nat.dist n m = n - m + (m - n)

nat.dist_comm (n m : ℕ) :
nat.dist n m = nat.dist m n

Attributes: simp
nat.dist_self (n : ℕ) :
nat.dist n n = 0

Attributes: simp
nat.eq_of_dist_eq_zero {n m : ℕ} (h : nat.dist n m = 0) :
n = m

nat.dist_eq_zero {n m : ℕ} (h : n = m) :
nat.dist n m = 0

nat.dist_eq_sub_of_le {n m : ℕ} (h : n ≤ m) :
nat.dist n m = m - n

nat.dist_eq_sub_of_ge {n m : ℕ} (h : n ≥ m) :
nat.dist n m = n - m

nat.dist_zero_right (n : ℕ) :
nat.dist n 0 = n

nat.dist_zero_left (n : ℕ) :
nat.dist 0 n = n

nat.dist_add_add_right (n k m : ℕ) :
nat.dist (n + k) (m + k) = nat.dist n m

nat.dist_add_add_left (k n m : ℕ) :
nat.dist (k + n) (k + m) = nat.dist n m

nat.dist_eq_intro {n m k l : ℕ} (h : n + m = k + l) :
nat.dist n k = nat.dist l m

nat.sub_lt_sub_add_sub (n m k : ℕ) :
n - k ≤ n - m + (m - k)

nat.dist.triangle_inequality (n m k : ℕ) :
nat.dist n k ≤ nat.dist n m + nat.dist m k

nat.dist_mul_right (n k m : ℕ) :
nat.dist (n * k) (m * k) = nat.dist n m * k

nat.dist_mul_left (k n m : ℕ) :
nat.dist (k * n) (k * m) = k * nat.dist n m

nat.dist_succ_succ {i j : ℕ} :
nat.dist (nat.succ i) (nat.succ j) = nat.dist i j

nat.dist_pos_of_ne {i j : ℕ} (a : i ≠ j) :
0 < nat.dist i j
