int.sum_two_squares_of_two_mul_sum_two_squares

∀ {m x y : ℤ}, 2 * m = x ^ 2 + y ^ 2 → m = ((x - y) / 2) ^ 2 + ((x + y) / 2) ^ 2

int.exists_sum_two_squares_add_one_eq_k

∀ {p : ℕ}, nat.prime p → (∃ (a b : ℤ) (k : ℕ), a ^ 2 + b ^ 2 + 1 = ↑k * ↑p ∧ k < p)

nat.sum_four_squares

∀ (n : ℕ), ∃ (a b c d : ℕ), a ^ 2 + b ^ 2 + c ^ 2 + d ^ 2 = n
