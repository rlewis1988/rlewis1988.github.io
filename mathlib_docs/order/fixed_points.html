set α

α

Least fixed point of a monotone function

lattice.gfp {α : Type u} [lattice.complete_lattice α] (f : α → α) :
α

Greatest fixed point of a monotone function

lattice.lfp_le {α : Type u} [lattice.complete_lattice α] {f : α → α} {a : α} (h : f a ≤ a) :
lattice.lfp f ≤ a

lattice.le_lfp {α : Type u} [lattice.complete_lattice α] {f : α → α} {a : α} (h : ∀ (b : α), f b ≤ b → a ≤ b) :
a ≤ lattice.lfp f

lattice.lfp_eq {α : Type u} [lattice.complete_lattice α] {f : α → α} (m : monotone f) :
lattice.lfp f = f (lattice.lfp f)

lattice.lfp_induct {α : Type u} [lattice.complete_lattice α] {f : α → α} {p : α → Prop} (m : monotone f) (step : ∀ (a : α), p a → a ≤ lattice.lfp f → p (f a)) (sup : ∀ (s : set α), (∀ (a : α), a ∈ s → p a) → p (lattice.Sup s)) :
p (lattice.lfp f)

lattice.monotone_lfp {α : Type u} [lattice.complete_lattice α] :
monotone lattice.lfp

lattice.le_gfp {α : Type u} [lattice.complete_lattice α] {f : α → α} {a : α} (h : a ≤ f a) :
a ≤ lattice.gfp f

lattice.gfp_le {α : Type u} [lattice.complete_lattice α] {f : α → α} {a : α} (h : ∀ (b : α), b ≤ f b → b ≤ a) :
lattice.gfp f ≤ a

lattice.gfp_eq {α : Type u} [lattice.complete_lattice α] {f : α → α} (m : monotone f) :
lattice.gfp f = f (lattice.gfp f)

lattice.gfp_induct {α : Type u} [lattice.complete_lattice α] {f : α → α} {p : α → Prop} (m : monotone f) (step : ∀ (a : α), p a → lattice.gfp f ≤ a → p (f a)) (inf : ∀ (s : set α), (∀ (a : α), a ∈ s → p a) → p (lattice.Inf s)) :
p (lattice.gfp f)

lattice.monotone_gfp {α : Type u} [lattice.complete_lattice α] :
monotone lattice.gfp

lattice.lfp_comp {α : Type u} {β : Type v} [lattice.complete_lattice α] [lattice.complete_lattice β] {f : β → α} {g : α → β} (m_f : monotone f) (m_g : monotone g) :
lattice.lfp (f ∘ g) = f (lattice.lfp (g ∘ f))

lattice.gfp_comp {α : Type u} {β : Type v} [lattice.complete_lattice α] [lattice.complete_lattice β] {f : β → α} {g : α → β} (m_f : monotone f) (m_g : monotone g) :
lattice.gfp (f ∘ g) = f (lattice.gfp (g ∘ f))

lattice.lfp_lfp {α : Type u} [lattice.complete_lattice α] {h : α → α → α} (m : ∀ ⦃a b c d : α⦄, a ≤ b → c ≤ d → h a c ≤ h b d) :
lattice.lfp (lattice.lfp ∘ h) = lattice.lfp (λ (x : α), h x x)

lattice.gfp_gfp {α : Type u} [lattice.complete_lattice α] {h : α → α → α} (m : ∀ ⦃a b c d : α⦄, a ≤ b → c ≤ d → h a c ≤ h b d) :
lattice.gfp (lattice.gfp ∘ h) = lattice.gfp (λ (x : α), h x x)

lattice.fixed_points.prev {α : Type u} [lattice.complete_lattice α] (f : α → α) (x : α) :
α

lattice.fixed_points.next {α : Type u} [lattice.complete_lattice α] (f : α → α) (x : α) :
α

lattice.fixed_points.prev_le {α : Type u} [lattice.complete_lattice α] {f : α → α} {x : α} :
lattice.fixed_points.prev f x ≤ x

lattice.fixed_points.prev_eq {α : Type u} [lattice.complete_lattice α] {f : α → α} (hf : monotone f) {a : α} (h : f a ≤ a) :
lattice.fixed_points.prev f a = f (lattice.fixed_points.prev f a)

lattice.fixed_points.prev_fixed {α : Type u} [lattice.complete_lattice α] {f : α → α} (hf : monotone f) (a : α) (h : f a ≤ a) :
↥(lattice.fixed_points f)

lattice.fixed_points.next_le {α : Type u} [lattice.complete_lattice α] {f : α → α} {x : α} :
x ≤ lattice.fixed_points.next f x

lattice.fixed_points.next_eq {α : Type u} [lattice.complete_lattice α] {f : α → α} (hf : monotone f) {a : α} (h : a ≤ f a) :
lattice.fixed_points.next f a = f (lattice.fixed_points.next f a)

lattice.fixed_points.next_fixed {α : Type u} [lattice.complete_lattice α] {f : α → α} (hf : monotone f) (a : α) (h : a ≤ f a) :
↥(lattice.fixed_points f)

lattice.fixed_points.sup_le_f_of_fixed_points {α : Type u} [lattice.complete_lattice α] (f : α → α) (hf : monotone f) (x y : ↥(lattice.fixed_points f)) :
x.val ⊔ y.val ≤ f (x.val ⊔ y.val)

lattice.fixed_points.f_le_inf_of_fixed_points {α : Type u} [lattice.complete_lattice α] (f : α → α) (hf : monotone f) (x y : ↥(lattice.fixed_points f)) :
f (x.val ⊓ y.val) ≤ x.val ⊓ y.val

lattice.fixed_points.Sup_le_f_of_fixed_points {α : Type u} [lattice.complete_lattice α] (f : α → α) (hf : monotone f) (A : set α) (HA : A ⊆ lattice.fixed_points f) :
lattice.Sup A ≤ f (lattice.Sup A)

lattice.fixed_points.f_le_Inf_of_fixed_points {α : Type u} [lattice.complete_lattice α] (f : α → α) (hf : monotone f) (A : set α) (HA : A ⊆ lattice.fixed_points f) :
f (lattice.Inf A) ≤ lattice.Inf A

lattice.fixed_points.lattice.complete_lattice {α : Type u} [lattice.complete_lattice α] (f : α → α) (hf : monotone f) :
lattice.complete_lattice ↥(lattice.fixed_points f)
