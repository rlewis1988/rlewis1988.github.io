span_gcd {α : Type u_1} [euclidean_domain α] (x y : α) :
ideal.span {euclidean_domain.gcd x y} = ideal.span {x, y}

gcd_is_unit_iff {α : Type u_1} [euclidean_domain α] {x y : α} :
is_unit (euclidean_domain.gcd x y) ↔ ideal.is_coprime x y
