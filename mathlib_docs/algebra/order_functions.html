strict_mono {α : Type u} {β : Type v} [has_lt α] [has_lt β] (f : α → β) :
Prop

A function f is strictly monotone if a < b implies f a < f b.

strict_mono.lt_iff_lt {α : Type u} {β : Type v} [linear_order α] [preorder β] {f : α → β} (H : strict_mono f) {a b : α} :
f a < f b ↔ a < b

ordering.compares o (f a) (f b) ↔ ordering.compares o a b

f a ≤ f b ↔ a ≤ b
