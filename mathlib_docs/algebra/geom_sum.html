geom_series {α : Type u} [semiring α] (x : α) (n : ℕ) :
α

Sum of the finite geometric series $∑_{i=0}^{n-1} x^i$.

geom_series_def {α : Type u} [semiring α] (x : α) (n : ℕ) :
geom_series x n = finset.sum (finset.range n) (λ (i : ℕ), x ^ i)

geom_series_zero {α : Type u} [semiring α] (x : α) :
geom_series x 0 = 0

geom_series₂_def {α : Type u} [semiring α] (x y : α) (n : ℕ) :
geom_series₂ x y n = finset.sum (finset.range n) (λ (i : ℕ), x ^ i * y ^ (n - 1 - i))

geom_sum_mul {α : Type u} [ring α] (x : α) (n : ℕ) :
geom_series x n * (x - 1) = x ^ n - 1

geom_sum_mul_neg {α : Type u} [ring α] (x : α) (n : ℕ) :
geom_series x n * (1 - x) = 1 - x ^ n

geom_sum {α : Type u} [division_ring α] {x : α} (h : x ≠ 1) (n : ℕ) :
geom_series x n = (x ^ n - 1) / (x - 1)
