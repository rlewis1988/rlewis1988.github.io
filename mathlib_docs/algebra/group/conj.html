is_conj {α : Type u} [group α] (a b : α) :
Prop

is_conj_refl {α : Type u} [group α] (a : α) :
is_conj a a

is_conj_symm {α : Type u} [group α] {a b : α} (a_1 : is_conj a b) :
is_conj b a

is_conj_trans {α : Type u} [group α] {a b c : α} (a_1 : is_conj a b) (a_2 : is_conj b c) :
is_conj a c

is_conj_one_right {α : Type u} [group α] {a : α} :
is_conj 1 a ↔ a = 1

Attributes: simp
is_conj_one_left {α : Type u} [group α] {a : α} :
is_conj a 1 ↔ a = 1

Attributes: simp
conj_inv {α : Type u} [group α] {a b : α} :
(b * a * b⁻¹)⁻¹ = b * a⁻¹ * b⁻¹

Attributes: simp
conj_mul {α : Type u} [group α] {a b c : α} :
b * a * b⁻¹ * (b * c * b⁻¹) = b * (a * c) * b⁻¹

Attributes: simp
is_conj_iff_eq {α : Type u_1} [comm_group α] {a b : α} :
is_conj a b ↔ a = b

Attributes: simp
is_group_hom.is_conj {α : Type u} {β : Type v} [group α] [group β] (f : α → β) [is_group_hom f] {a b : α} (a_1 : is_conj a b) :
is_conj (f a) (f b)
