is_group_anti_hom {α : Type u_1} {β : Type u_2} [group α] [group β] (f : α → β) :
Prop

Predicate for group anti-homomorphism, or a homomorphism into the opposite group.

is_group_anti_hom.map_one {α : Type u_1} {β : Type u_2} [group α] [group β] (f : α → β) [w : is_group_anti_hom f] :
f 1 = 1

is_group_anti_hom.map_inv {α : Type u_1} {β : Type u_2} [group α] [group β] (f : α → β) [w : is_group_anti_hom f] (a : α) :
f a⁻¹ = (f a)⁻¹

inv_is_group_anti_hom {α : Type u_1} [group α] :
is_group_anti_hom (λ (x : α), x⁻¹)
