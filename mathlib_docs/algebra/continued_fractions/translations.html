Translation Lemmas Between Functions Defined for Continued Fractions

Summary

Some simple translation lemmas between the different functions defined in algebra.continued_fractions.basic.

generalized_continued_fraction.terminated_at_iff_s_terminated_at

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, generalized_continued_fraction.terminated_at g n ↔ seq.terminated_at (g.s) n

generalized_continued_fraction.terminated_at_iff_s_none

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, generalized_continued_fraction.terminated_at g n ↔ seq.nth (g.s) n = option.none

generalized_continued_fraction.part_num_none_iff_s_none

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, seq.nth (generalized_continued_fraction.partial_numerators g) n = option.none ↔ seq.nth (g.s) n = option.none

generalized_continued_fraction.terminated_at_iff_part_num_none

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, generalized_continued_fraction.terminated_at g n ↔ seq.nth (generalized_continued_fraction.partial_numerators g) n = option.none

generalized_continued_fraction.part_denom_none_iff_s_none

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, seq.nth (generalized_continued_fraction.partial_denominators g) n = option.none ↔ seq.nth (g.s) n = option.none

generalized_continued_fraction.terminated_at_iff_part_denom_none

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ}, generalized_continued_fraction.terminated_at g n ↔ seq.nth (generalized_continued_fraction.partial_denominators g) n = option.none

generalized_continued_fraction.part_num_eq_s_a

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} {gp : generalized_continued_fraction.pair α}, seq.nth (g.s) n = option.some gp → seq.nth (generalized_continued_fraction.partial_numerators g) n = option.some (gp.a)

generalized_continued_fraction.part_denom_eq_s_b

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} {gp : generalized_continued_fraction.pair α}, seq.nth (g.s) n = option.some gp → seq.nth (generalized_continued_fraction.partial_denominators g) n = option.some (gp.b)

generalized_continued_fraction.obtain_s_a_of_part_num

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} {a : α}, seq.nth (generalized_continued_fraction.partial_numerators g) n = option.some a → (∃ (gp : generalized_continued_fraction.pair α), seq.nth (g.s) n = option.some gp ∧ gp.a = a)

generalized_continued_fraction.obtain_s_b_of_part_denom

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} {b : α}, seq.nth (generalized_continued_fraction.partial_denominators g) n = option.some b → (∃ (gp : generalized_continued_fraction.pair α), seq.nth (g.s) n = option.some gp ∧ gp.b = b)

generalized_continued_fraction.nth_cont_eq_succ_nth_cont_aux

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α], generalized_continued_fraction.continuants g n = generalized_continued_fraction.continuants_aux g (n + 1)

generalized_continued_fraction.num_eq_conts_a

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α], generalized_continued_fraction.numerators g n = (generalized_continued_fraction.continuants g n).a

generalized_continued_fraction.denom_eq_conts_b

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α], generalized_continued_fraction.denominators g n = (generalized_continued_fraction.continuants g n).b

generalized_continued_fraction.convergent_eq_num_div_denom

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α], generalized_continued_fraction.convergents g n = generalized_continued_fraction.numerators g n / generalized_continued_fraction.denominators g n

generalized_continued_fraction.convergent_eq_conts_a_div_conts_b

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α], generalized_continued_fraction.convergents g n = (generalized_continued_fraction.continuants g n).a / (generalized_continued_fraction.continuants g n).b

generalized_continued_fraction.obtain_conts_a_of_num

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α] {A : α}, generalized_continued_fraction.numerators g n = A → (∃ (conts : generalized_continued_fraction.pair α), generalized_continued_fraction.continuants g n = conts ∧ conts.a = A)

generalized_continued_fraction.obtain_conts_b_of_denom

∀ {α : Type u_1} {g : generalized_continued_fraction α} {n : ℕ} [_inst_1 : division_ring α] {B : α}, generalized_continued_fraction.denominators g n = B → (∃ (conts : generalized_continued_fraction.pair α), generalized_continued_fraction.continuants g n = conts ∧ conts.b = B)

generalized_continued_fraction.zeroth_continuant_aux_eq_one_zero

∀ {α : Type u_1} {g : generalized_continued_fraction α} [_inst_1 : division_ring α], generalized_continued_fraction.continuants_aux g 0 = {a := 1, b := 0}

Attributes: simp

generalized_continued_fraction.zeroth_convergent_eq_h
