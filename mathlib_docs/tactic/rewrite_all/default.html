tactic.all_rewrites

list (expr × bool) → expr → opt_param tactic.rewrite_all.cfg {to_rewrite_cfg := {to_apply_cfg := {md := tactic.transparency.semireducible, approx := bool.tt, new_goals := tactic.new_goals.non_dep_first, instances := bool.tt, auto_param := bool.tt, opt_param := bool.tt, unify := bool.tt}, symm := bool.ff, occs := occurrences.all}, try_simp := bool.ff, discharger := tactic.skip, simplifier := λ (e : expr), tactic.failed} → tactic.mllist tactic (tactic.rewrite_all.tracked_rewrite × ℕ × ℕ)

return a lazy list of (t, n, k) where * t is a tracked_rewrite (i.e. a pair (e' : expr, prf : e = e')) * n is the index of the rule r used from rs, and * k is the index of t in all_rewrites r e.
