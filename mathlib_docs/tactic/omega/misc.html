omega.fun_mono_2

∀ {α β γ : Type} {p : α → β → γ} {a1 a2 : α} {b1 b2 : β}, a1 = a2 → b1 = b2 → p a1 b1 = p a2 b2

omega.pred_mono_2

∀ {α β : Type} {p : α → β → Prop} {a1 a2 : α} {b1 b2 : β}, a1 = a2 → b1 = b2 → (p a1 b1 ↔ p a2 b2)

omega.pred_mono_2'

∀ {c : Prop → Prop → Prop} {a1 a2 b1 b2 : Prop}, (a1 ↔ a2) → (b1 ↔ b2) → (c a1 b1 ↔ c a2 b2)

omega.update

Π {α : Type}, ℕ → α → (ℕ → α) → ℕ → α

omega.update_eq

∀ {α : Type} (m : ℕ) (a : α) (v : ℕ → α), omega.update m a v m = a

omega.update_eq_of_ne

∀ {α : Type} {m : ℕ} {a : α} {v : ℕ → α} (k : ℕ), k ≠ m → omega.update m a v k = v k

omega.update_zero

Π {α : Type}, α → (ℕ → α) → ℕ → α
