omega.lin_comb

list ℕ → list omega.term → omega.term

Attributes: simp

omega.lin_comb_holds

∀ {v : ℕ → ℤ} {ts : list omega.term} (ns : list ℕ), (∀ (t : omega.term), t ∈ ts → 0 ≤ omega.term.val v t) → 0 ≤ omega.term.val v (omega.lin_comb ns ts)

omega.unsat_lin_comb

list ℕ → list omega.term → Prop

omega.unsat_lin_comb_of

∀ (ns : list ℕ) (ts : list omega.term), (omega.lin_comb ns ts).fst < 0 → (∀ (x : ℤ), x ∈ (omega.lin_comb ns ts).snd → x = 0) → omega.unsat_lin_comb ns ts

omega.unsat_of_unsat_lin_comb

∀ (ns : list ℕ) (ts : list omega.term), omega.unsat_lin_comb ns ts → omega.clause.unsat (list.nil omega.term, ts)
