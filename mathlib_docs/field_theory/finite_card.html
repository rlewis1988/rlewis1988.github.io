finite_field.card (α : Type u) [discrete_field α] [fintype α] (p : ℕ) [char_p α p] :
∃ (n : ℕ+), nat.prime p ∧ fintype.card α = p ^ ↑n

finite_field.card' (α : Type u) [discrete_field α] [fintype α] :
∃ (p : ℕ) (n : ℕ+), nat.prime p ∧ fintype.card α = p ^ ↑n
