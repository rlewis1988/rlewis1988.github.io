polynomial.splits {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] (f : polynomial α) :
Prop

a polynomial splits iff it is zero or all of its irreducible factors have degree 1

polynomial.splits_zero {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] :
polynomial.splits i 0

Attributes: simp
polynomial.splits_C {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] (a : α) :
polynomial.splits i (polynomial.C a)

polynomial.splits_mul {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f g : polynomial α} (hf : polynomial.splits i f) (hg : polynomial.splits i g) :

polynomial.splits i f ∧ polynomial.splits i g

polynomial.splits_map_iff {α : Type u} {β : Type v} {γ : Type w} [discrete_field α] [discrete_field β] [discrete_field γ] (i : α → β) [is_field_hom i] (j : β → γ) [is_field_hom j] {f : polynomial α} :
polynomial.splits j (polynomial.map i f) ↔ polynomial.splits (λ (x : α), j (i x)) f

polynomial.exists_root_of_splits {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f : polynomial α} (hs : polynomial.splits i f) (hf0 : polynomial.degree f ≠ 0) :
∃ (x : β), polynomial.eval₂ i x f = 0

polynomial.exists_multiset_of_splits {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f : polynomial α} (a : polynomial.splits i f) :
∃ (s : multiset β), polynomial.map i f = polynomial.C (i (polynomial.leading_coeff f)) * multiset.prod (multiset.map (λ (a : β), polynomial.X - polynomial.C a) s)

polynomial.splits_of_exists_multiset {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f : polynomial α} {s : multiset β} (hs : polynomial.map i f = polynomial.C (i (polynomial.leading_coeff f)) * multiset.prod (multiset.map (λ (a : β), polynomial.X - polynomial.C a) s)) :
polynomial.splits i f

polynomial.splits_of_splits_id {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f : polynomial α} (a : polynomial.splits id f) :

polynomial.splits_iff_exists_multiset {α : Type u} {β : Type v} [discrete_field α] [discrete_field β] (i : α → β) [is_field_hom i] {f : polynomial α} :
polynomial.splits i f ↔ ∃ (s : multiset β), polynomial.map i f = polynomial.C (i (polynomial.leading_coeff f)) * multiset.prod (multiset.map (λ (a : β), polynomial.X - polynomial.C a) s)
