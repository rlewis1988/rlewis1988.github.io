monotonicity

(user_attribute unit)

monotonicity.pi

∀ {α : Sort u} {p q : α → Prop}, (∀ (a : α), implies (p a) (q a)) → implies (∀ (a : α), p a) (∀ (a : α), q a)

monotonicity.imp

∀ {p p' q q' : Prop}, implies p' q' → implies q p → implies (p → p') (q → q')

monotonicity.const

∀ (p : Prop), implies p p

monotonicity.true

∀ (p : Prop), implies p true

monotonicity.false

∀ (p : Prop), implies false p

monotonicity.exists

∀ {α : Sort u} {p q : α → Prop}, (∀ (a : α), implies (p a) (q a)) → implies (∃ (a : α), p a) (∃ (a : α), q a)

monotonicity.and

∀ {p p' q q' : Prop}, implies p p' → implies q q' → implies (p ∧ q) (p' ∧ q')

monotonicity.or

∀ {p p' q q' : Prop}, implies p p' → implies q q' → implies (p ∨ q) (p' ∨ q')

monotonicity.not

∀ {p q : Prop}, implies p q → implies (¬q) (¬p)
