category_theory.left_adjoint_preserves_epi

∀ {C : Type u₁} [𝒞 : category_theory.category C] {D : Type u₂} [𝒟 : category_theory.category D] {F : C ⥤ D} {G : D ⥤ C}, (F ⊣ G) → ∀ {X Y : C} {f : X ⟶ Y}, category_theory.epi f → category_theory.epi (F.map f)

category_theory.faithful_reflects_epi

∀ {C : Type u₁} [𝒞 : category_theory.category C] {D : Type u₂} [𝒟 : category_theory.category D] (F : C ⥤ D) [_inst_1 : category_theory.faithful F] {X Y : C} {f : X ⟶ Y}, category_theory.epi (F.map f) → category_theory.epi f
