Conjugate morphisms by isomorphisms

An isomorphism α : X ≅ Y defines - a monoid isomorphism conj : End X ≃* End Y by α.conj f = α.inv ≫ f ≫ α.hom; - a group isomorphism conj_Aut : Aut X ≃* Aut Y by α.conj_Aut f = α.symm ≪≫ f ≪≫ α.

For completeness, we also define hom_congr : (X ≅ X₁) → (Y ≅ Y₁) → (X ⟶ Y) ≃ (X₁ ⟶ Y₁), cf. equiv.arrow_congr.

category_theory.iso.hom_congr

Π {C : Type u} [𝒞 : category_theory.category C] {X Y X₁ Y₁ : C}, (X ≅ X₁) → (Y ≅ Y₁) → (X ⟶ Y) ≃ (X₁ ⟶ Y₁)

category_theory.iso.hom_congr_apply

∀ {C : Type u} [𝒞 : category_theory.category C] {X Y X₁ Y₁ : C} (α : X ≅ X₁) (β : Y ≅ Y₁) (f : X ⟶ Y), ⇑(category_theory.iso.hom_congr α β) f = α.inv ≫ f ≫ β.hom

category_theory.iso.hom_congr_comp

∀ {C : Type u} [𝒞 : category_theory.category C] {X Y Z X₁ Y₁ Z₁ : C} (α : X ≅ X₁) (β : Y ≅ Y₁) (γ : Z ≅ Z₁) (f : X ⟶ Y) (g : Y ⟶ Z), ⇑(category_theory.iso.hom_congr α γ) (f ≫ g) = ⇑(category_theory.iso.hom_congr α β) f ≫ ⇑(category_theory.iso.hom_congr β γ) g

category_theory.iso.hom_congr_refl

∀ {C : Type u} [𝒞 : category_theory.category C] {X Y : C} (f : X ⟶ Y), ⇑(category_theory.iso.hom_congr (category_theory.iso.refl X) (category_theory.iso.refl Y)) f = f

∀ {C : Type u} [𝒞 : category_theory.category C] {X₁ Y₁ X₂ Y₂ X₃ Y₃ : C} (α₁ : X₁ ≅ X₂) (β₁ : Y₁ ≅ Y₂) (α₂ : X₂ ≅ X₃) (β₂ : Y₂ ≅ Y₃) (f : X₁ ⟶ Y₁), ⇑(category_theory.iso.hom_congr (α₁ ≪≫ α₂) (β₁ ≪≫ β₂)) f = ⇑(equiv.trans (category_theory.iso.hom_congr α₁ β₁) (category_theory.iso.hom_congr α₂ β₂)) f
