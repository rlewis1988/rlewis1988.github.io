bitvec.bits_to_nat_to_list

∀ {n : ℕ} (x : bitvec n), bitvec.to_nat x = bitvec.bits_to_nat (vector.to_list x)

bitvec.to_nat_append

∀ {m : ℕ} (xs : bitvec m) (b : bool), bitvec.to_nat (vector.append xs (b :: vector.nil)) = bitvec.to_nat xs * 2 + bitvec.to_nat (b :: vector.nil)

bitvec.bits_to_nat_to_bool

∀ (n : ℕ), bitvec.to_nat (decidable.to_bool (n % 2 = 1) :: vector.nil) = n % 2

bitvec.of_nat_succ

∀ {k n : ℕ}, bitvec.of_nat (nat.succ k) n = vector.append (bitvec.of_nat k (n / 2)) (decidable.to_bool (n % 2 = 1) :: vector.nil)

bitvec.to_nat_of_nat

∀ {k n : ℕ}, bitvec.to_nat (bitvec.of_nat k n) = n % 2 ^ k
