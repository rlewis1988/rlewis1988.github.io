lazy_list α

lazy_list α

lazy_list.to_list {α : Type u} (a : lazy_list α) :
list α

lazy_list.head {α : Type u} [inhabited α] (a : lazy_list α) :

lazy_list.tail {α : Type u} (a : lazy_list α) :
lazy_list α
