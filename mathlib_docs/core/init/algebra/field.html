structure division_ring :

  • zero : α

  • add_left_neg : ∀ (a : α), -a + a = 0
  • add_comm : ∀ (a b : α), a + b = b + a
  • mul : α → α → α
  • mul_assoc : ∀ (a b c_1 : α), a * b * c_1 = a * (b * c_1)
  • one : α
  • one_mul : ∀ (a : α), 1 * a = a
  • mul_one : ∀ (a : α), a * 1 = a

  • inv : α → α
  • zero_ne_one : 0 ≠ 1
  • mul_inv_cancel : ∀ {a : α}, a ≠ 0 → a * a⁻¹ = 1
  • inv_mul_cancel : ∀ {a : α}, a ≠ 0 → a⁻¹ * a = 1
