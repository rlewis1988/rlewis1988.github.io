classical.choice {α : Sort u} (a : nonempty α) :
α

classical.indefinite_description {α : Sort u} (p : α → Prop) (h : ∃ (x : α), p x) :
{x // p x}

classical.some {α : Sort u} {p : α → Prop} (h : ∃ (x : α), p x) :
α

classical.some_spec {α : Sort u} {p : α → Prop} (h : ∃ (x : α), p x) :
p (classical.some h)

classical.em (p : Prop) :
p ∨ ¬p

classical.exists_true_of_nonempty {α : Sort u} (a : nonempty α) :
∃ (x : α), true

classical.inhabited_of_nonempty {α : Sort u} (h : nonempty α) :
inhabited α

classical.inhabited_of_exists {α : Sort u} {p : α → Prop} (h : ∃ (x : α), p x) :
inhabited α

classical.prop_decidable (a : Prop) :
decidable a

classical.decidable_inhabited (a : Prop) :
inhabited (decidable a)

classical.type_decidable_eq (α : Sort u) :
decidable_eq α

classical.type_decidable (α : Sort u) :
psum α (α → false)

classical.strong_indefinite_description {α : Sort u} (p : α → Prop) (h : nonempty α) :
{x // (∃ (y : α), p y) → p x}

classical.epsilon {α : Sort u} [h : nonempty α] (p : α → Prop) :
α

classical.epsilon_spec_aux {α : Sort u} (h : nonempty α) (p : α → Prop) (a : ∃ (y : α), p y) :
p (classical.epsilon p)

classical.epsilon_spec {α : Sort u} {p : α → Prop} (hex : ∃ (y : α), p y) :
p (classical.epsilon p)

classical.epsilon_singleton {α : Sort u} (x : α) :
classical.epsilon (λ (y : α), y = x) = x

classical.axiom_of_choice {α : Sort u} {β : α → Sort v} {r : Π (x : α), β x → Prop} (h : ∀ (x : α), ∃ (y : β x), r x y) :
∃ (f : Π (x : α), β x), ∀ (x : α), r x (f x)

classical.skolem {α : Sort u} {b : α → Sort v} {p : Π (x : α), b x → Prop} :
(∀ (x : α), ∃ (y : b x), p x y) ↔ ∃ (f : Π (x : α), b x), ∀ (x : α), p x (f x)

classical.prop_complete (a : Prop) :
a = true ∨ a = false

classical.eq_true_or_eq_false (a : Prop) :
a = true ∨ a = false

classical.cases_true_false (p : Prop → Prop) (h1 : p true) (h2 : p false) (a : Prop) :
p a

classical.cases_on (a : Prop) {p : Prop → Prop} (h1 : p true) (h2 : p false) :
p a

classical.by_cases {p q : Prop} (hpq : p → q) (hnpq : ¬p → q) :
q

classical.by_contradiction {p : Prop} (h : ¬p → false) :
p

classical.eq_false_or_eq_true (a : Prop) :
a = false ∨ a = true
