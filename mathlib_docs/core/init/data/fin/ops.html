fin.add_def {n : ℕ} (a b : fin n) :
(a + b).val = (a.val + b.val) % n

fin.mul_def {n : ℕ} (a b : fin n) :
(a * b).val = a.val * b.val % n

fin.sub_def {n : ℕ} (a b : fin n) :
(a - b).val = a.val - b.val

fin.mod_def {n : ℕ} (a b : fin n) :
(a % b).val = a.val % b.val

fin.div_def {n : ℕ} (a b : fin n) :
(a / b).val = a.val / b.val

fin.lt_def {n : ℕ} (a b : fin n) :
a < b = (a.val < b.val)

fin.le_def {n : ℕ} (a b : fin n) :
a ≤ b = (a.val ≤ b.val)

fin.val_zero {n : ℕ} :
0.val = 0
