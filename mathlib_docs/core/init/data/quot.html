inductive eqv_gen {α : Type u} :
(α → α → Prop) → α → α → Prop
  • rel : ∀ {α : Type u} (r : α → α → Prop) (a a_1 : α), r a a_1 → eqv_gen r a a_1
  • refl : ∀ {α : Type u} (r : α → α → Prop) (a : α), α → eqv_gen r a a
  • symm : ∀ {α : Type u} (r : α → α → Prop) (a a_1 : α), eqv_gen r a a_1 → eqv_gen r a_1 a
  • trans : ∀ {α : Type u} (r : α → α → Prop) (a a_1 z : α), eqv_gen r a a_1 → eqv_gen r a_1 z → eqv_gen r a z
