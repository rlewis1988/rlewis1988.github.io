subtype.exists_of_subtype {α : Type u} {p : α → Prop} (a : {x // p x}) :
∃ (x : α), p x

subtype.tag_irrelevant {α : Type u} {p : α → Prop} {a : α} (h1 h2 : p a) :
⟨a, h1⟩ = ⟨a, h2⟩

subtype.eq {α : Type u} {p : α → Prop} {a1 a2 : {x // p x}} (a : a1.val = a2.val) :
a1 = a2

subtype.eta {α : Type u} {p : α → Prop} (a : {x // p x}) (h : p (a.val)) :
⟨a.val, h⟩ = a
