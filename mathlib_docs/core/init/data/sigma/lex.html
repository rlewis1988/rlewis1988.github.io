psigma.lex

Π {α : Sort u} {β : α → Sort v}, (α → α → Prop) → (Π (a : α), β a → β a → Prop) → psigma β → psigma β → Prop

psigma.lex_accessible

∀ {α : Sort u} {β : α → Sort v} {r : α → α → Prop} {s : Π (a : α), β a → β a → Prop} {a : α}, acc r a → (∀ (a : α), well_founded (s a)) → ∀ (b : β a), acc (psigma.lex r s) ⟨a, b⟩

psigma.lex_wf

∀ {α : Sort u} {β : α → Sort v} {r : α → α → Prop} {s : Π (a : α), β a → β a → Prop}, well_founded r → (∀ (x : α), well_founded (s x)) → well_founded (psigma.lex r s)

psigma.lex_ndep_wf

∀ {α : Sort u} {β : Sort v} {r : α → α → Prop} {s : β → β → Prop}, well_founded r → well_founded s → well_founded (psigma.lex_ndep r s)

∀ {α : Sort u} {β : Sort v} {r : α → α → Prop} {s : β → β → Prop} {b : β}, acc s b → (∀ (a : α), acc r a) → ∀ (a : α), acc (psigma.rev_lex r s) ⟨a, b⟩
