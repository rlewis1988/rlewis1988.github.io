array.slice {α : Type u} {n : ℕ} (a : array n α) (k l : ℕ) (h₁ : k ≤ l) (h₂ : l ≤ n) :
array (l - k) α

array.take {α : Type u} {n : ℕ} (a : array n α) (m : ℕ) (h : m ≤ n) :
array m α

array.drop {α : Type u} {n : ℕ} (a : array n α) (m : ℕ) (h : m ≤ n) :
array (n - m) α

array.take_right {α : Type u} {n : ℕ} (a : array n α) (m : ℕ) (h : m ≤ n) :
array m α

array.reverse {α : Type u} {n : ℕ} (a : array n α) :
array n α
