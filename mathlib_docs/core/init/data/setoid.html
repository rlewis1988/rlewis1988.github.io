setoid (α : Sort u) :
Sort (max 1 u)

setoid_has_equiv {α : Sort u} [setoid α] :
has_equiv α

Attributes: instance
setoid.refl {α : Sort u} [setoid α] (a : α) :
a ≈ a

setoid.symm {α : Sort u} [setoid α] {a b : α} (hab : a ≈ b) :
b ≈ a

setoid.trans {α : Sort u} [setoid α] {a b c : α} (hab : a ≈ b) (hbc : b ≈ c) :
a ≈ c
