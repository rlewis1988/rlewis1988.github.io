option.is_lawful_monad

is_lawful_monad option

Attributes: instance

option.eq_of_eq_some

∀ {α : Type u} {x y : option α}, (∀ (z : α), x = option.some z ↔ y = option.some z) → x = y

option.eq_some_of_is_some

∀ {α : Type u} {o : option α} (h : ↥(option.is_some o)), o = option.some (option.get h)

option.eq_none_of_is_none

∀ {α : Type u} {o : option α}, ↥(option.is_none o) → o = option.none
