def rbnode.ins {α : Type u} (lt : α → α → Prop) [decidable_rel lt] :
rbnode α → α → rbnode α

Equations
  • rbnode.ins lt (rbnode.black_node a y b) x = rbnode.ins._match_2 a y b x (rbnode.ins lt a x) (rbnode.ins lt b x) (cmp_using lt x y)
  • rbnode.ins lt (rbnode.red_node a y b) x = rbnode.ins._match_1 a y b x (rbnode.ins lt a x) (rbnode.ins lt b x) (cmp_using lt x y)
  • rbnode.ins lt rbnode.leaf x = rbnode.red_node rbnode.leaf x rbnode.leaf
  • rbnode.ins._match_2 a y b x _f_1 _f_2 ordering.gt = ite (rbnode.get_color b = rbnode.color.red) (rbnode.balance2_node _f_2 y a) (rbnode.black_node a y _f_2)
  • rbnode.ins._match_2 a y b x _f_1 _f_2 ordering.eq = rbnode.black_node a x b
  • rbnode.ins._match_2 a y b x _f_1 _f_2 ordering.lt = ite (rbnode.get_color a = rbnode.color.red) (rbnode.balance1_node _f_1 y b) (rbnode.black_node _f_1 y b)
  • rbnode.ins._match_1 a y b x _f_1 _f_2 ordering.gt = rbnode.red_node a y _f_2
  • rbnode.ins._match_1 a y b x _f_1 _f_2 ordering.eq = rbnode.red_node a x b
  • rbnode.ins._match_1 a y b x _f_1 _f_2 ordering.lt = rbnode.red_node _f_1 y b
