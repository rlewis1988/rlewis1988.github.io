complex.exists_forall_abs_polynomial_eval_le (p : polynomial ℂ) :
∃ (x : ℂ), ∀ (y : ℂ), complex.abs (polynomial.eval x p) ≤ complex.abs (polynomial.eval y p)

complex.exists_root {f : polynomial ℂ} (hf : 0 < polynomial.degree f) :
∃ (z : ℂ), polynomial.is_root f z
