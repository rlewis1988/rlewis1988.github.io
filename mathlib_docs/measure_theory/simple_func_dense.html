measure_theory.simple_func_sequence_tendsto {α : Type u} {β : Type v} [measure_theory.measure_space α] [normed_group β] [topological_space.second_countable_topology β] {f : α → β} (hf : measurable f) :
∃ (F : ℕ → measure_theory.simple_func α β), ∀ (x : α), filter.tendsto (λ (n : ℕ), ⇑(F n) x) filter.at_top (nhds (f x)) ∧ ∀ (n : ℕ), ∥⇑(F n) x∥ ≤ ∥f x∥ + ∥f x∥
