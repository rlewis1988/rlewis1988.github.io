function.embedding.schroeder_bernstein {α : Type u} {β : Type v} {f : α → β} {g : β → α} (hf : function.injective f) (hg : function.injective g) :
∃ (h : α → β), function.bijective h

function.embedding.antisymm {α : Type u} {β : Type v} (a : α ↪ β) (a_1 : β ↪ α) :
nonempty (α ≃ β)

function.embedding.injective_min {ι : Type u} {β : ι → Type v} (I : nonempty ι) :
∃ (i : ι), nonempty (Π (j : ι), β i ↪ β j)

function.embedding.total {α : Type u} {β : Type v} :
nonempty (α ↪ β) ∨ nonempty (β ↪ α)
