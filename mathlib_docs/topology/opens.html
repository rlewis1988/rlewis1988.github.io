topological_space.opens.is_basis B ↔ ∀ {U : topological_space.opens α} {x : α}, x ∈ U → (∃ (U' : topological_space.opens α) (H : U' ∈ B), x ∈ U' ∧ U' ⊆ U)

topological_space.opens.is_basis B ↔ ∀ (U : topological_space.opens α), ∃ (Us : set (topological_space.opens α)) (H : Us ⊆ B), U = lattice.Sup Us
