polynomial.tendsto_infinity {α : Type u_1} {β : Type u_2} [comm_ring α] [discrete_linear_ordered_field β] (abv : α → β) [is_absolute_value abv] {p : polynomial α} (h : 0 < polynomial.degree p) (x : β) :
∃ (r : β) (H : r > 0), ∀ (z : α), r < abv z → x < abv (polynomial.eval z p)
