list.topological_space {α : Type u_1} [topological_space α] :
topological_space (list α)

Attributes: instance
nhds_list {α : Type u_1} [topological_space α] (as : list α) :
nhds as = traversable.traverse nhds as

nhds_nil {α : Type u_1} [topological_space α] :
nhds list.nil = has_pure.pure list.nil

nhds_cons {α : Type u_1} [topological_space α] (a : α) (l : list α) :
nhds (a :: l) = list.cons <$> nhds a <*> nhds l

list.tendsto_cons' {α : Type u_1} [topological_space α] {a : α} {l : list α} :
filter.tendsto (λ (p : α × list α), p.fst :: p.snd) (filter.prod (nhds a) (nhds l)) (nhds (a :: l))

list.tendsto_cons {β : Type u_2} [topological_space β] {α : Type u_1} {f : α → β} {g : α → list β} {a : filter α} {b : β} {l : list β} (hf : filter.tendsto f a (nhds b)) (hg : filter.tendsto g a (nhds l)) :
filter.tendsto (λ (a : α), f a :: g a) a (nhds (b :: l))

list.tendsto_cons_iff {α : Type u_1} [topological_space α] {β : Type u_2} {f : list α → β} {b : filter β} {a : α} {l : list α} :
filter.tendsto f (nhds (a :: l)) b ↔ filter.tendsto (λ (p : α × list α), f (p.fst :: p.snd)) (filter.prod (nhds a) (nhds l)) b
